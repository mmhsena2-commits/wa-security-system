<script>
let progress = 0;
const validNumber = "083168453133";

let logs = [
"[✓] Menghubungkan ke server...",
"[✓] Validasi nomor...",
"[✓] Scan sistem keamanan...",
"[✓] Sinkronisasi perangkat...",
"[✓] Finalisasi sistem..."
];

function start() {
    let input = document.getElementById("phone").value.trim();

    document.getElementById("output").innerHTML = "";
    document.getElementById("success").style.display = "none";
    document.getElementById("failed").style.display = "none";
    document.getElementById("bar").style.width = "0%";
    document.getElementById("percent").innerText = "0%";

    progress = 0;
    logs = [
        "[✓] Menghubungkan ke server...",
        "[✓] Validasi nomor...",
        "[✓] Scan sistem keamanan...",
        "[✓] Sinkronisasi perangkat...",
        "[✓] Finalisasi sistem..."
    ];

    if (input !== validNumber) {
        document.getElementById("output").innerHTML = "[!] Nomor tidak terdaftar dalam sistem.\nProses dihentikan.";
        document.getElementById("failed").style.display = "block";
        return;
    }

    runProgress();
}

function runProgress() {
    if (progress <= 100) {
        document.getElementById("bar").style.width = progress + "%";
        document.getElementById("percent").innerText = progress + "%";

        if (progress % 20 === 0 && logs.length > 0) {
            document.getElementById("output").innerHTML += logs.shift() + "\n";
        }

        progress++;
        setTimeout(runProgress, 600);
    } else {
        document.getElementById("output").innerHTML += "\n[✓] Sistem selesai.";
        document.getElementById("success").style.display = "block";
    }
}

function contactAdmin() {
    let nomor = document.getElementById("phone").value.trim();
    const message = encodeURIComponent(
        "Halo admin, saya ingin konfirmasi hasil pengerjaan untuk nomor " + nomor + 
        ". Saya siap melakukan pelunasan pembayaran akhir. Mohon info langkah selanjutnya."
    );
    window.open("https://wa.me/6285716070218?text=" + message, "_blank");
}
